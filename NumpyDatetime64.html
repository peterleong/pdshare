# Changing the X-axis date xticks using the matplotlib dates locator API
# https://matplotlib.org/api/dates_api.html?highlight=date%20locator#matplotlib.dates.YearLocator
import numpy as np
import matplotlib.pyplot as plt
import matplotlib.dates as mdates

yloc = mdates.YearLocator() # set to every year major axis
fig,ax = plt.subplots(figsize=(8,5))
plt.plot_date(X,y,"g-")
ax.xaxis.set_major_locator(yloc) 
fig.autofmt_xdate()
plt.show()
